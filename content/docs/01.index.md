## 部署脚本

::code-collapse

```bash [init-env.sh]
#!/bin/bash
set -e

echo ">>> 部署 Java..."
mkdir -p /opt/java
tar -zxvf openjdk*.tar.gz -C /opt/java --strip-components=1
/opt/java/bin/java -version
echo ">>> Java 部署完成"

echo ">>> 部署 Docker..."
tar -zxvf docker*.tgz -C /usr/local/bin --strip-components=1
cat <<EOF > /etc/systemd/system/docker.service
[Unit]
Description=Docker service
After=containerd.service
[Service]
ExecStart=/usr/local/bin/dockerd
Type=notify
Delegate=yes
KillMode=process
Restart=always
RestartSec=5
LimitNPROC=infinity
LimitCORE=infinity
TasksMax=infinity
OOMScoreAdjust=-999
[Install]
WantedBy=multi-user.target
EOF

systemctl daemon-reload
systemctl enable docker
systemctl start docker

docker -v
echo ">>> Docker 部署完成"

echo ">>> 导入 Nginx 镜像..."
NGINX_TAR=$(ls nginx*.tar | head -n1)
echo $NGINX_TAR
NGINX_IMAGE_NAME=$(docker load < $NGINX_TAR | awk -F': ' '/Loaded image:/ {print $2}')
echo "Nginx 镜像 $NGINX_IMAGE_NAME 导入完成"
docker images
echo ">>> Nginx 镜像导入完成"

echo ">>> Nginx 容器启动..."
mkdir -p /opt/nginx/conf /opt/nginx/html
cat <<EOF > /opt/nginx/conf/default.conf
server {
    listen 80;
    server_name localhost;

    root /usr/share/nginx/html;
    index index.html;

    location / {
        try_files \$uri \$uri/ =404;
    }
}
EOF

cat <<EOF > /opt/nginx/html/index.html
<!DOCTYPE html>
<html>
<head>
    <meta charset="UTF-8">
    <title>Docker Nginx 测试</title>
</head>
<body>
    <h1>Hello, Nginx in Docker!</h1>
    <p>访问成功！</p>
</body>
</html>
EOF

NGINX_CONTAINER_NAME="nginx"
if docker ps -a --format '{{.Names}}' | grep -Eq "^$NGINX_CONTAINER_NAME\$"; then
    echo ">>> Nginx 容器已存在，停止并删除..."
    docker stop $NGINX_CONTAINER_NAME
    docker rm $NGINX_CONTAINER_NAME
fi
docker run -d --name $NGINX_CONTAINER_NAME -v /opt/nginx/conf:/etc/nginx/conf.d -v /opt/nginx/html:/usr/share/nginx/html:ro --network host $NGINX_IMAGE_NAME
echo ">>> 启动 Nginx 容器完成"

docker ps
echo "http://127.0.0.1:80"

echo ">>> 部署 Rustfs..."
mkdir -p /opt/rustfs/bin /opt/rustfs/conf /data/rustfs0 /var/logs/rustfs /opt/tls
chmod -R 750 /data/rustfs* /var/logs/rustfs
unzip -o rustfs*.zip -d /opt/rustfs/bin
chmod +x /opt/rustfs/bin/rustfs

cat <<EOF > /opt/rustfs/conf/default.conf
RUSTFS_ACCESS_KEY=hero
RUSTFS_SECRET_KEY=City@2026
RUSTFS_VOLUMES="/data/rustfs0"
RUSTFS_ADDRESS="0.0.0.0:6000"
RUSTFS_CONSOLE_ADDRESS="0.0.0.0:6001"
RUSTFS_CONSOLE_ENABLE=true
RUST_LOG=error
RUSTFS_OBS_LOG_DIRECTORY="/var/logs/rustfs/"
EOF

#if [ ! -f /etc/systemd/system/rustfs.service ]; then
cat <<EOF > /etc/systemd/system/rustfs.service
[Unit]
Description=RustFS Object Storage Server
Documentation=https://rustfs.com/docs/
After=network-online.target
Wants=network-online.target
[Service]
Type=notify
NotifyAccess=main
User=root
Group=root
WorkingDirectory=/opt/rustfs
EnvironmentFile=-/opt/rustfs/conf/default.conf
ExecStart=/opt/rustfs/bin/rustfs $RUSTFS_VOLUMES
LimitNOFILE=1048576
LimitNPROC=32768
TasksMax=infinity
Restart=always
RestartSec=10s
OOMScoreAdjust=-1000
SendSIGKILL=no
TimeoutStartSec=30s
TimeoutStopSec=30s
NoNewPrivileges=true
ProtectSystem=full
ProtectHome=true
PrivateTmp=true
PrivateDevices=true
ProtectClock=true
ProtectKernelTunables=true
ProtectKernelModules=true
ProtectControlGroups=true
RestrictSUIDSGID=true
RestrictRealtime=true
[Install]
WantedBy=multi-user.target
EOF

systemctl daemon-reload
systemctl enable rustfs
systemctl start rustfs
#fi

/opt/rustfs/bin/rustfs --version
echo ">>> Rustfs 部署完成"

echo ">>> ttyd 部署..."
mkdir -p /opt/ttyd/bin
chmod +x ttyd.*
cp -rf ttyd.* /opt/ttyd/bin/ttyd

cat <<EOF > /etc/systemd/system/ttyd.service
[Unit]
Description=ttyd Web Terminal
After=network-online.target
Wants=network-online.target

[Service]
Type=simple
User=root
Group=root
WorkingDirectory=/opt/ttyd
ExecStart=/opt/ttyd/bin/ttyd  -p 6100 -W -c hero:City@2026 /bin/bash
Restart=always
RestartSec=1
LimitNOFILE=65536
MemoryLimit=512M
CPUQuota=100%

[Install]
WantedBy=multi-user.target
EOF

systemctl daemon-reload
systemctl enable ttyd
systemctl start ttyd

/opt/ttyd/bin/ttyd --version
echo ">>> ttyd 部署完成"

echo ">>> 导入 Cloudbeaver 镜像..."
CLOUDBEAVER_TAR=$(ls dbeaver-cloudbeaver*.tar | head -n1)
echo $CLOUDBEAVER
CLOUDBEAVER_IMAGE_NAME=$(docker load < $CLOUDBEAVER_TAR | awk -F': ' '/Loaded image:/ {print $2}')
echo "Cloudbeaver 镜像 $CLOUDBEAVER_IMAGE_NAME 导入完成"
docker images
echo ">>> Cloudbeaver 镜像导入完成"

echo ">>> Cloudbeaver 容器启动"
CLOUDBEAVER_CONTAINER_NAME="cloudbeaver"
if docker ps -a --format '{{.Names}}' | grep -Eq "^$CLOUDBEAVER_CONTAINER_NAME\$"; then
    echo ">>> Cloudbeaver 容器已存在，停止并删除..."
    docker stop $CLOUDBEAVER_CONTAINER_NAME
    docker rm $CLOUDBEAVER_CONTAINER_NAME
fi
docker run --name $CLOUDBEAVER_CONTAINER_NAME -d --restart unless-stopped --network host -v /opt/cloudbeaver/workspace:/opt/cloudbeaver/workspace $CLOUDBEAVER_IMAGE_NAME
echo ">>> Cloudbeaver 容器启动完成"

docker ps
echo "http://127.0.0.1:8978"

echo ">>> 部署 Nacos..."
CONF_FILE=/opt/nacos/conf/application.properties
TOKEN_SECRET_KEY="VGhpc0lzTXlDdXN0b21TZWNyZXRLZXkwMTIzNDU2Nzg="
IDENTITY_KEY="hero"
IDENTITY_VALUE="City@2026"
unzip -o nacos*.zip -d /opt
# 确保文件末尾有换行
sed -i -e '$a\' "$CONF_FILE"

# 删除旧配置（包括注释、空值）
sed -i \
  -e '/^#\?\s*nacos\.core\.auth\.plugin\.nacos\.token\.secret\.key=/d' \
  -e '/^#\?\s*nacos\.core\.auth\.server\.identity\.key=/d' \
  -e '/^#\?\s*nacos\.core\.auth\.server\.identity\.value=/d' \
  "$CONF_FILE"

# 追加正确配置
cat <<EOF >> "$CONF_FILE"
nacos.core.auth.plugin.nacos.token.secret.key=${TOKEN_SECRET_KEY}
nacos.core.auth.server.identity.key=${IDENTITY_KEY}
nacos.core.auth.server.identity.value=${IDENTITY_VALUE}
EOF

#if [ ! -f /etc/systemd/system/nacos.service ]; then
cat <<EOF > /etc/systemd/system/nacos.service
[Unit]
Description=Nacos Service
After=network-online.target
Wants=network-online.target
[Service]
Type=forking
WorkingDirectory=/opt/nacos
Environment=JAVA_HOME=/opt/java
ExecStart=/bin/bash /opt/nacos/bin/startup.sh -m standalone
ExecStop=/bin/bash /opt/nacos/bin/shutdown.sh
Restart=on-failure
RestartSec=10
LimitNOFILE=65535
User=root
Group=root
[Install]
WantedBy=multi-user.target
EOF

systemctl daemon-reload
systemctl enable nacos
systemctl start nacos
#fi

echo ">>> Nacos 部署完成"

echo ">>> 部署 MySQL..."
MYSQL_BASE=/opt/mysql
MYSQL_DATA=$MYSQL_BASE/data
MYSQL_LOG=$MYSQL_BASE/log
MYSQL_CONF=$MYSQL_BASE/conf
MYSQL_PORT=6306
MYSQL_USER=mysql
MYSQL_ROOT_PASS=City@2026
echo ">>> 创建 mysql 用户（若不存在）"
id $MYSQL_USER >/dev/null 2>&1 || useradd -r -s /sbin/nologin $MYSQL_USER
echo ">>> 创建目录"
mkdir -p $MYSQL_BASE $MYSQL_DATA $MYSQL_LOG $MYSQL_CONF
chown -R $MYSQL_USER:$MYSQL_USER $MYSQL_BASE
echo ">>> 解压 MySQL"
if [ ! -f "$MYSQL_BASE/bin/mysqld" ]; then
    tar -xf mysql*.tar.xz -C $MYSQL_BASE --strip-components=1
    echo ">>> MySQL 解压完成"
else
    echo ">>> MySQL 已存在，跳过解压"
fi
echo ">>> 写 my.cnf"
cat > $MYSQL_CONF/my.cnf <<EOF
[mysqld]
basedir=$MYSQL_BASE
datadir=$MYSQL_DATA
socket=$MYSQL_BASE/run/mysql.sock
pid-file=$MYSQL_BASE/run/mysqld.pid
port=$MYSQL_PORT
bind-address=0.0.0.0
log-error=$MYSQL_LOG/mysql.err
slow_query_log=1
slow_query_log_file=$MYSQL_LOG/slow.log
character-set-server=utf8mb4
collation-server=utf8mb4_general_ci
skip-name-resolve
max_connections=500
[client]
socket=$MYSQL_BASE/run/mysql.sock
EOF
chown $MYSQL_USER:$MYSQL_USER $MYSQL_CONF/my.cnf
if [ ! -d "$MYSQL_DATA/mysql" ]; then
    mkdir -p $MYSQL_BASE/run
    chown -R $MYSQL_USER:$MYSQL_USER $MYSQL_BASE/run
    $MYSQL_BASE/bin/mysqld \
        --defaults-file=$MYSQL_CONF/my.cnf \
        --initialize \
        --user=$MYSQL_USER \
        2> $MYSQL_LOG/init.log
fi
ROOT_TEMP_PASS=$(grep "temporary password" $MYSQL_LOG/mysql.err | awk '{print $NF}')
echo $ROOT_TEMP_PASS
echo ">>> 写 systemd 服务"
if [ ! -f /etc/systemd/system/mysql.service ]; then
cat >/etc/systemd/system/mysql.service <<EOF
[Unit]
Description=MySQL Server
After=network.target

[Service]
Type=notify
User=$MYSQL_USER
Group=$MYSQL_USER
ExecStart=$MYSQL_BASE/bin/mysqld --defaults-file=$MYSQL_CONF/my.cnf
ExecStop=$MYSQL_BASE/bin/mysqladmin shutdown --defaults-file=$MYSQL_CONF/my.cnf
Restart=always
LimitNOFILE=65536

[Install]
WantedBy=multi-user.target
EOF

systemctl daemon-reload
systemctl enable mysql
systemctl start mysql

sleep 5

echo ">>> MySQL 启动状态"
systemctl status mysql --no-pager

echo ">>> 配置 root@'%' 可远程访问"
$MYSQL_BASE/bin/mysql -uroot -p"$ROOT_TEMP_PASS" \
    --socket=$MYSQL_BASE/run/mysql.sock --connect-expired-password <<SQL
ALTER USER 'root'@'localhost' IDENTIFIED BY '$MYSQL_ROOT_PASS';
CREATE USER IF NOT EXISTS 'root'@'%' IDENTIFIED BY '$MYSQL_ROOT_PASS';
GRANT ALL PRIVILEGES ON *.* TO 'root'@'%' WITH GRANT OPTION;
FLUSH PRIVILEGES;
SQL
fi

echo
echo "=========================================="
echo " MySQL 部署完成"
echo " MySQL 端口: $MYSQL_PORT"
echo " MySQL root 密码: $MYSQL_ROOT_PASS"
echo
echo " 登录命令示例（本机或远程）:"
echo " $MYSQL_BASE/bin/mysql -uroot -p$MYSQL_ROOT_PASS -h <ip> --socket=$MYSQL_BASE/run/mysql.sock"

echo ">>> Keycloak 部署完成"
```
